// --- [1] 數據核心：加入明細存儲結構 --- let db = JSON.parse(localStorage.getItem('PX_EN_V15.0')) || { name: "Miaoyi", totalNian: 0, bodhiLedger: [] // 每筆紀錄包含 {id, amount, items: [], date} }; function save() { localStorage.setItem('PX_EN_V15.0', JSON.stringify(db)); } // --- [2] 萬用法門：發票明細解析器 --- const InvoiceParser = { // 解析左側碼 parseSummary: function(raw) { if (raw.length < 50 || raw.startsWith("**")) return null; try { const amount = parseInt(raw.substring(29, 37), 16); const invNum = raw.substring(0, 10); return { invNum: invNum, amount: amount, date: (parseInt(raw.substring(10, 13)) + 1911) + "-" + raw.substring(13, 15) + "-" + raw.substring(15, 17) }; } catch(e) { return null; } }, // 解析右側碼 (明細) parseDetails: function(raw) { if (!raw.startsWith("**")) return null; try { // 移除前綴 ** 並根據冒號分割 // 格式通常為：品名:數量:單價:品名:數量:單價... const content = raw.substring(2); const parts = content.split(":"); let items = []; for (let i = 0; i < parts.length; i += 3) { if (parts[i]) { items.push({ name: parts[i].trim(), qty: parts[i+1] || "1", price: parts[i+2] || "0" }); } } return items; } catch(e) { return null; } } }; // --- [3] 掃描控制中心：兩段式注入 --- let lastInvNum = null; // 暫存剛掃到的發票號碼，用來對接明細 async function handleScanResult(raw, status) { // 優先判斷是否為明細 (右側碼) if (raw.startsWith("**")) { const details = InvoiceParser.parseDetails(raw); if (details && db.bodhiLedger.length > 0) { // 尋找最後一筆紀錄（或比對發票號碼）進行注入 db.bodhiLedger[0].items = details; save(); updateUI(); if (navigator.vibrate) navigator.vibrate([30, 100]); status.innerText = "✨ Details Injected! Record Complete."; setTimeout(closeScanner, 1000); } else { status.innerText = "Please scan the LEFT QR first."; } } // 否則判斷是否為總金額 (左側碼) else { const summary = InvoiceParser.parseSummary(raw); if (summary) { // 建立新紀錄，放在最前面 db.bodhiLedger.unshift({ id: Date.now(), invNum: summary.invNum, amount: summary.amount, date: summary.date, cat: "General", items: [], // 等待右側碼掃描 color: "var(--accent-gold)" }); save(); updateUI(); if (navigator.vibrate) navigator.vibrate(50); status.innerText = `Captured $${summary.amount}! Now scan the RIGHT code.`; status.style.color = "var(--accent-jade)"; } } } // --- [4] UI 顯化：蓮花綻放式清單 --- function updateUI() { const total = db.bodhiLedger.reduce((acc, t) => acc + t.amount, 0); const balanceDisplay = document.getElementById('main-balance'); if(balanceDisplay) balanceDisplay.innerText = "$" + total.toLocaleString(); const listObj = document.getElementById('ledger-list'); if(!listObj) return; listObj.innerHTML = db.bodhiLedger.slice(0, 10).map(t => `
${t.date} ${t.invNum || ''}
-$${t.amount.toLocaleString()}
${t.items.length > 0 ? t.items.map(i => `
${i.name} x${i.qty} $${i.price}
`).join('') : '
No details (Scan right QR to add)
'}
`).join(''); }
